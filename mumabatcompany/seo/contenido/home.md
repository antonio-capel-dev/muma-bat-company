import { Helmet } from "react-helmet-async"

export default function HeadHome() {
  const schema = {
    "@context": "https://schema.org",
    "@graph": [
      {
        "@type": "Organization",
        "name": "MUMA BAT COMPANY",
        "url": "https://mumabatcompany.com",
        "description": "Empresa especializada en servicios con murciélagos: conservacion, control biologico, educacion ambiental, realidad virtual y consultoria.",
        "areaServed": "ES",
        "memberOf": [
          { "@type": "Organization", "name": "SECEMU" },
          { "@type": "Organization", "name": "EUROBATS" },
          { "@type": "Organization", "name": "REV" }
        ]
      },
      {
        "@type": "WebSite",
        "name": "MUMA BAT COMPANY",
        "url": "https://mumabatcompany.com"
      }
    ]
  }

  return (
    <Helmet>
      <title>MUMA BAT COMPANY | Servicios con murciélagos para empresas</title>
      <meta name="description" content="Experiencias inmersivas, refugios, educacion ambiental y consultoria basada en la naturaleza. Para ayuntamientos, museos, colegios y empresas." />
      <link rel="canonical" href="https://mumabatcompany.com/" />
      <script type="application/ld+json">{JSON.stringify(schema)}</script>
    </Helmet>
  )
}

## 9. ADVERTENCIAS SEO Y DECISIONES ESTRATEGICAS

[CRITICO] Title y metadatos
  El title actual no tiene intencion comercial. Reemplazar obligatoriamente.
  Sin react-helmet-async no hay metadatos por pagina. Instalar primero.
  npm install react-helmet-async + envolver App.jsx con HelmetProvider.

[CRITICO] CTAs de cards apuntan a #contacto
  Deben apuntar a las landings de servicio una vez creadas.
  Sin esto Google no entiende la jerarquia del sitio.

[IMPORTANTE] H1 actual sin intencion de busqueda
  Reemplazar por el H1 propuesto o la alternativa segun decision de marca.

[IMPORTANTE] Estadisticas de credibilidad
  Validar cifras con el cliente antes de publicar.
  Si son limitadas, sustituir por avales institucionales (SECEMU, EUROBATS, REV).

[IMPORTANTE] Mapa de refugios
  El mapa Leaflet actual es un placeholder sin datos reales.
  No publicar con datos ficticios. Eliminar o sustituir hasta tener datos.

[IMPORTANTE] Seccion proyectos en la home
  Solo publicar si hay proyectos con datos verificables y fichas creadas en /proyectos.
  Los proyectos identificados (Parador Golf, Bat Night, ST3ER) son reales
  pero sus fichas aun no existen. Mientras tanto: solo enlace a /proyectos.

[MENOR] Numero de WhatsApp
  El placeholder wa.me/34600000000 debe sustituirse por el numero real.

[MENOR] URL del dominio definitivo
  Confirmar la URL definitiva antes de implementar canonicals y schemas.

[MENOR] Canibalizacion
  No hay riesgo si se mantiene la separacion: home = semicomercial/branded,
  landings = transaccional especifico por servicio.
  Vigilar: no crear articulos de blog con el mismo H1 que las landings.
